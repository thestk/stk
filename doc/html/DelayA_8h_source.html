00125 inline StkFloat DelayA :: nextOut( void )
00126 {
00127   if ( doNextOut_ ) {
00128     // Do allpass interpolation delay.
00129     nextOutput_ = -coeff_ * lastFrame_[0];
00130     nextOutput_ += apInput_ + ( coeff_ * inputs_[outPoint_] );
