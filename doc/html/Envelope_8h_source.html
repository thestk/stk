00088 inline StkFloat Envelope :: tick( void )
00089 {
00090   if ( state_ ) {
00091     if ( target_ > value_ ) {
00092       value_ += rate_;
00093       if ( value_ >= target_ ) {
00094         value_ = target_;
00095         state_ = 0;
00096       }
00097     }
00098     else {
00099       value_ -= rate_;
00100       if ( value_ <= target_ ) {
00101         value_ = target_;
00102         state_ = 0;
00103       }
00104     }
00105     lastFrame_[0] = value_;
00106   }
00107 
00108   return value_;
00109 }
