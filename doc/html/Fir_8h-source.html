00086 inline StkFloat Fir :: tick( StkFloat input )
00087 {
00088   lastFrame_[0] = 0.0;
00089   inputs_[0] = gain_ * input;
00090 
00091   for ( unsigned int i=b_.size()-1; i>0; i-- ) {
00092     lastFrame_[0] += b_[i] * inputs_[i];
